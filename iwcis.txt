const readline = require("readline");

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

rl.question("Enter first number: ", (first) => {
  rl.question("Enter second number: ", (second) => {
    const sum = parseInt(first) + parseInt(second);
    console.log("Sum:", sum);
    rl.close();
  });
});
